<template>
  <div class="contact-form-container">
    <h2>{{ $t('pages.contact.form_title') }}</h2>
    <form>
      <div class="group-container">
        <div class="form-group" v-for="part in parts" :class="{ active: part.isActive }">
          <label :for="part.name">{{ $t(part.label) }} {{ part.required ? '*' : '' }}:</label>
          <input :id="part.name" :type="part.type" :placeholder="$t(part.placeholder)" :required="part.required" :pattern="part.pattern" v-model="contact[part.name]" />
        </div>
      </div>
      <b-row>
        <b-col v-if="button.prev.isActive">
          <div class="text-left">
            <b-button type="button" class="round-btn" @click="prevForm()">
              <i class="ti ti-lg ti-arrow-left"></i> {{ $t('actions.prev') }}
            </b-button>
          </div>
        </b-col>
        <b-col>
          <div class="text-right">
            <b-button type="submit" class="round-btn" @click="nextForm()">
              {{ $t(button.next.text) }} <i class="ti ti-lg ti-arrow-right"></i>
            </b-button>
          </div>
        </b-col>
      </b-row>
    </form>
  </div>
</template>
 
<script>
  export default {
    name: 'ContactFormComponent',
    data() {
      return {
        parts: [{
          name: 'name',
          type: 'text',
          placeholder: 'pages.contact.placeholder.name',
          required: true,
          isActive: true,
          label: 'pages.contact.label.name',
        }, {
          name: 'company',
          type: 'text',
          placeholder: 'pages.contact.placeholder.company',
          required: false,
          isActive: false,
          label: 'pages.contact.label.company',
        }, {
          name: 'email',
          type: 'email',
          placeholder: 'pages.contact.placeholder.email',
          required: true,
          isActive: false,
          label: 'pages.contact.label.email',
          pattern: /^.+@\S+\.\S+$/,
        }, {
          name: 'phone',
          type: 'text',
          placeholder: 'pages.contact.placeholder.phone',
          required: false,
          isActive: false,
          label: 'pages.contact.label.phone',
        }, {
          name: 'message',
          type: 'text',
          placeholder: 'pages.contact.placeholder.message',
          required: true,
          isActive: false,
          label: 'pages.contact.label.message',
        }],
        contact: {
          name: '',
          company: '',
          email: '',
          phone: '',
          message: '',
        },
        button: {
          next: {
            text: 'actions.next',
          },
          prev: {
            isActive: false,
          },
        },
      };
    },
    methods: {
      send() {
        this.$http.post(this.api('public/contact/'), {
          contact: this.contact,
        }).then((result) => {
          if (result.data.status) {
            this.$notify({ title: 'title', text: this.$t('flashes.contact.sent'), type: 'success' });
          } else {
            this.$notify({ title: 'title', text: this.$t('flashes.contact.not_sent'), type: 'error' });
          }
        }, () => {
          this.$notify({ title: 'title', text: this.$t('flashes.contact.server_error'), type: 'error' });
        });
        return false;
      },
      prevForm() {
        _.some(this.parts, (part, idx, parts) => {
          const ret = part.isActive === true;
          if (ret) {
            part.isActive = false;
 
            let index = idx - 1;
            if (idx === 1) {
              this.$set(this.button.prev, 'isActive', false);
              index = 0;
            }
 
            parts[index].isActive = true;
            this.$set(parts, index, parts[index]);
            this.$set(parts, idx, part);
          }
          return ret;
        });
      },
      nextForm() {
        _.some(this.parts, (part, idx, parts) => {
          const ret = part.isActive === true;
          if (part.required && this.contact[part.name] === '') { return false; }
          if (ret) {
            part.isActive = false;
 
            let index = idx + 1;
            if (idx === 0) {
              this.$set(this.button.prev, 'isActive', true);
            }
 
            if (index === parts.length - 1) {
              this.$set(this.button.next, 'text', 'actions.send');
            }
 
            if (index === parts.length) {
              this.send();
              this.$parent.close();
              this.$set(this.contact, 'name', '');
              this.$set(this.contact, 'company', '');
              this.$set(this.contact, 'email', '');
              this.$set(this.contact, 'phone', '');
              this.$set(this.contact, 'message', '');
              this.$set(this.button.next, 'text', 'actions.next');
              this.$set(this.button.prev, 'isActive', false);
              index = 0;
            }
 
            parts[index].isActive = true;
            this.$set(parts, index, parts[index]);
            this.$set(parts, idx, part);
          }
          return ret;
        });
      },
    },
  };
</script>
